# 一个有趣的并查集
## 并查集+2Sat思想


给定长n的01串s, 给定k个集合A1,...,Ak,保证任意三个集合交集为空. 每次操作选择一个集合,翻转s中对应位置. 定义mi为使前i个位置全为1所需的最少操作数(题目数据保证每个mi都存在), 求所有mi的值.


```c++
int n,m;

int fa[N],siz[N];
int find(int x){
	if(x==fa[x])return x;
	return fa[x]=find(fa[x]);
}
void unin(int a,int b){
	a=find(a),b=find(b);
	if(a==b)return;
	fa[a]=b;siz[b]+=siz[a];
}
int sel(int x){
	return min(siz[find(x)],siz[find(x+m)]);
}

char c[N];
vector<int>vec[N];
void solv(){
	n=read(),m=read();	
	for(int i=1;i<=m*2;i++){
		fa[i]=i;
		if(i<=m)siz[i]=1;
		else siz[i]=0;
	}
	siz[m*2+1]=1e9;
	fa[m*2+1]=m*2+1;
	for(int i=1;i<=n;i++){
		vec[i].clear();
	}
	scanf("%s",c+1);
	for(int i=1;i<=m;i++){
		int r=read();
		for(int j=1;j<=r;j++){
			int id=read();
			vec[id].push_back(i);
		}
	}
	int nowans=0;
	for(int i=1;i<=n;i++){
		if(vec[i].size()==1){
			int ax=vec[i][0];
			nowans-=sel(ax);
			if(c[i]=='0'){
				unin(ax+m,m*2+1);
			}else{
				unin(ax,m*2+1);
			}
			nowans+=sel(ax);
		}else if(vec[i].size()==2){
			int a=vec[i][0],b=vec[i][1];
			if(find(a)==find(b)||find(a+m)==find(b)){
				
			}else{
				nowans-=sel(a)+sel(b);
				if(c[i]=='0'){
					unin(a+m,b);unin(a,b+m);
				}else{
					unin(a,b);unin(a+m,b+m);
				}
				nowans+=sel(a);
			}
		}
		printf("%d\n",nowans);
	}	
}
```
