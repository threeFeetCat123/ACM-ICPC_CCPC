# 逆序对
归并排序
O(nlongn)
```c++

ll cul(int c[],int l,int r){
	if(l==r)return 0;
	int mid=l+r>>1;
	int i=l,j=mid+1;
	int k=l-1;
	ll ans=cul2(c,l,mid)+cul2(c,mid+1,r);
	while(i<=mid&&j<=r){
		if(c[i]<=c[j]){
			tmp[++k]=c[i];
			i++;
		}else{
			tmp[++k]=c[j];
			j++;
			ans+=mid-i+1;
		}
	}
	while(i<=mid){
		tmp[++k]=c[i];
		i++;
	}
	while(j<=r){
		tmp[++k]=c[j];
		j++;
	}
	
	for(int i=l;i<=r;i++){
		c[i]=tmp[i];
	}
	
	return ans;
	
}
```
