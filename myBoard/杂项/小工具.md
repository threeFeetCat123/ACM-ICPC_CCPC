# int128 输出流
```c++
using i128 = __int128;

std::ostream &operator<<(std::ostream &os, i128 n) {
    std::string s;
    while (n) {
        s += '0' + n % 10;
        n /= 10;
    }
    std::reverse(s.begin(), s.end());
    return os << s;
}
```

# 手写堆
```c++
struct Heap{
	ll q[N], cnt;
	void pushup(int id) {
		while (id > 1) {
			if (q[id] >= q[id >> 1]) break;
			swap(q[id], q[id >> 1]);
			id >>= 1;
		}
	}
	void movedown() {
		int id = 1;
		while (id << 1 <= cnt) {
			if ((id << 1 | 1) <= cnt) {
				if (q[id] < min(q[id << 1], q[id << 1 | 1])) break;;
				if (q[id << 1] < q[id << 1 | 1]) swap(q[id], q[id << 1]), id <<= 1;
				else swap(q[id], q[id << 1 | 1]), id = id << 1 | 1;
			}
			else {
				if (q[id] > q[id << 1]) swap(q[id], q[id << 1]);
				break;
			}
		}
	}
	void push(ll x) {
		q[++cnt] = x;
		pushup(cnt);
	}
	void pop() {
		swap(q[1], q[cnt]);
		cnt--;
		movedown();
	}
	ll top(){
		return q[1];
	}
}pq;
```
