异或线性基
struct LineBase{
	int r;
	ll a[N];
	LineBase(int _r=62){
		r=_r;
		for(int i=0;i<=r;i++)a[i]=0;
	}
	void insert(ll x){
		for(int i=r;i>=0&&x;i--){
			if(x&(1ll<<i)){
				if(a[i]){
					x^=a[i];
				}else{
					a[i]=x;
					break;
				}
			}
		}
	}
	void mergeFrom(const LineBase b){
		for(int i=0;i<=r;i++)insert(b.a[i]);
	}
	ll query(ll x){
		for(int i=r;i>=0;i--){
			if((x^a[i])>x)x^=a[i];
		}
		return x;
	}
}lb;
