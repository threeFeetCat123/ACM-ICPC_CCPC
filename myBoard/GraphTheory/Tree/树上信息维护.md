## 树上信息维护

### 子树加、树链加、子树查、树链查
#### P3384 【模板】重链剖分/树链剖分

#### 题目描述

如题，已知一棵包含 $N$ 个结点的树（连通且无环），每个节点上包含一个数值，需要支持以下操作：

- `1 x y z`，表示将树从 $x$ 到 $y$ 结点最短路径上所有节点的值都加上 $z$。

- `2 x y`，表示求树从 $x$ 到 $y$ 结点最短路径上所有节点的值之和。

- `3 x z`，表示将以 $x$ 为根节点的子树内所有节点值都加上 $z$。

- `4 x` 表示求以 $x$ 为根节点的子树内所有节点值之和


```c++
//树上信息维护 
#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
const int N=1e5+9;
int mod=1e9+7;

int n,m,rot;
ll w[N];
//重链剖分 
vector<int>l[N];
int dep[N],siz[N],wson[N],f[N];
int top[N],dfn[N][2],dfncnt,invdfn[N];
void dfs_size(int now,int fa){
	f[now]=fa;
	siz[now]=1;
	dep[now]=dep[fa]+1;
	for(auto to:l[now]){
		if(to==fa)continue;
		dfs_size(to,now);
		siz[now]+=siz[to];
		if(siz[wson[now]]<siz[to])wson[now]=to;
	}
}
void dfs_top(int now,int tp){
	top[now]=tp;
	dfn[now][0]=++dfncnt;
	invdfn[dfncnt]=now;
	if(wson[now]){
		dfs_top(wson[now],tp);
	}
	for(auto to:l[now]){
		if(to==f[now]||to==wson[now])continue;
		dfs_top(to,to);
	}
	dfn[now][1]=dfncnt;
}

//区加区和线段树 
struct Point{
	int l,r;
	ll sum,lazy;
}p[N<<2];
void pushup(int now){
	p[now].sum=p[now<<1].sum+p[now<<1|1].sum;
}
void downlazy(int now){
	if(p[now].lazy==0)return;
	int lson=now<<1,rson=now<<1|1;
	p[lson].lazy+=p[now].lazy;p[rson].lazy+=p[now].lazy;
	p[lson].sum+=p[now].lazy*(p[lson].r-p[lson].l+1);
	p[rson].sum+=p[now].lazy*(p[rson].r-p[rson].l+1);
	p[now].lazy=0;
}
void build(int now,int l,int r){
	p[now].l=l;p[now].r=r;
	p[now].lazy=0;
	if(l==r){
		p[now].sum=w[invdfn[l]];
		return;
	}
	int mid=l+r>>1;
	build(now<<1,l,mid);
	build(now<<1|1,mid+1,r);
	pushup(now);
}
void change(int now,int l,int r,ll k){
	if(l<=p[now].l&&p[now].r<=r){
		p[now].lazy+=k;
		p[now].sum+=k*(p[now].r-p[now].l+1);
		return;
	}
	downlazy(now);
	int mid=p[now].l+p[now].r>>1;
	if(l<=mid){
		change(now<<1,l,r,k);
	}
	if(r>mid){
		change(now<<1|1,l,r,k);
	}
	pushup(now);
}
ll query(int now,int l,int r){
	if(l<=p[now].l&&p[now].r<=r){
		return p[now].sum;
	}
	downlazy(now);
	int mid=p[now].l+p[now].r>>1;
	ll res=0;
	if(l<=mid){
		res+=query(now<<1,l,r);
	}
	if(r>mid){
		res+=query(now<<1|1,l,r);
	}
	pushup(now);
	return res;
}

//树上操作
void add(int l,int r,ll k){//子树加+单点加 
	change(1,l,r,k);
}
void addlca(int a,int b,ll k){//两点路径加 
	while(top[a]!=top[b]){
		if(dep[top[a]]<dep[top[b]]){
			swap(a,b);
		}
		add(dfn[top[a]][0],dfn[a][0],k);
		a=f[top[a]];
	}
	if(dep[a]<dep[b]){
		swap(a,b);
	}
	add(dfn[b][0],dfn[a][0],k);
}
ll queson(int a){//子树求和 
	return query(1,dfn[a][0],dfn[a][1]);
} 
ll que(int a,int b){//两点路径求和 
	ll res=0;
	while(top[a]!=top[b]){
		if(dep[top[a]]<dep[top[b]]){
			swap(a,b);
		}
		res+=query(1,dfn[top[a]][0],dfn[a][0]);
		a=f[top[a]];
	}
	if(dep[a]<dep[b]){
		swap(a,b);
	}
	res+=query(1,dfn[b][0],dfn[a][0]);
	return res;
}
void solv(){
	cin>>n>>m>>rot>>mod;
	for(int i=1;i<=n;i++){
		cin>>w[i];
	}
	for(int i=1;i<n;i++){
		int a,b;
		cin>>a>>b;
		l[a].push_back(b);
		l[b].push_back(a);
	}
	dfs_size(rot,0);
	dfs_top(rot,rot);
	build(1,1,n);
	for(int i=1;i<=m;i++){
		int opt;
		cin>>opt;
		if(opt==1){//树链加
			int a,b;ll k;
			cin>>a>>b>>k;
			addlca(a,b,k);
		}else if(opt==2){//树链查 
			int a,b;
			cin>>a>>b;
			cout<<que(a,b)%mod<<'\n';
		}else if(opt==3){//子树加
			int a;ll k;
			cin>>a>>k;
			add(dfn[a][0],dfn[a][1],k);
		}else if(opt==4){//子树查 
			int a;
			cin>>a;
			cout<<query(1,dfn[a][0],dfn[a][1])%mod<<'\n';
		}else{
			cout<<"cin WA\n";
			return;
		}
	}
}

int main(){
	std::ios::sync_with_stdio(false);
	cin.tie(0); 
	cout.tie(0);
	int tst=1;
//	cin>>tst;
	while(tst--){
		solv();
	}
}
```
