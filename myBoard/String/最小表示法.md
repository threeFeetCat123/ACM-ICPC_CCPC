int ans[maxn];
void min_rp(int c[],int len){
	for(int i=1;i<=len;i++){
		c[len+i]=c[i];
	}
	int l=1,r=2,k=0;
	for(int i=1;l<=len&&k<=len&&r<=len;i++){
		int opt=c[l+k]-c[r+k];
		if(opt==0){
			k++;continue;
		}
		else if(opt>0){
			l+=k+1;
		}else{
			r+=k+1;
		}
		if(l==r)l++;
		k=0;
	}
	l=min(l,r);
	r=l+len;
	for(int i=l;i<=r;i++){
		ans[i-l+1]=c[i];
	}
}
int n;
int c[maxn];
int main(){
	n=read();
	for(int i=1;i<=n;i++){
		c[i]=read();
	}
	min_rp(c,n);
	for(int i=1;i<=n;i++){
		printf("%d ",ans[i]);
	}
	return 0;
}
